                        currentWeight = parseFloat(weights[i]);

                    }

                }


                const totalLbsLost = startWeight - currentWeight;

                const totalPctLost = startWeight > 0 ? (totalLbsLost / startWeight) * 100 : 0;

            };

                    let bestLbs = -Infinity;

                        if (currentVal === undefined || currentVal === null || currentVal === '') return;

                        const prevVal = w === 1 ? p.startingWeight : weights[w - 1];

                        if (prevVal === undefined || prevVal === null || prevVal === '') return;

                        const lbsLost = prevVal - currentVal;

                        const pctLost = (lbsLost / prevVal) * 100;


                        if (pctLost > bestPct || (pctLost === bestPct && lbsLost > bestLbs)) {

                            bestPct = pctLost;

                            bestLbs = lbsLost;

                            winner = { name: p.name, pct: pctLost };

                        }

                    });

                    if (winner) winners.push({ week: w, ...winner });

                }
